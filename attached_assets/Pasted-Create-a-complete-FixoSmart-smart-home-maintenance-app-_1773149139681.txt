### 6. 🗄️ SUPABASE SETUP

```sql
-- Create all tables with multilingual support

CREATE TABLE profiles (
  id UUID REFERENCES auth.users PRIMARY KEY,
  full_name TEXT,
  phone TEXT,
  language TEXT DEFAULT 'en', -- 'bn', 'en', 'ar'
  iqama_expiry DATE,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE services (
  id SERIAL PRIMARY KEY,
  name_bn TEXT,  -- Bangla name
  name_en TEXT,  -- English name
  name_ar TEXT,  -- Arabic name
  category TEXT,
  price_sar DECIMAL,
  image_url TEXT
);

CREATE TABLE products (
  id SERIAL PRIMARY KEY,
  name_bn TEXT,
  name_en TEXT,
  name_ar TEXT,
  price_sar DECIMAL,
  installation_fee_sar DECIMAL DEFAULT 99,
  affiliate_link TEXT,
  image_url TEXT
);

CREATE TABLE bookings (
  id SERIAL PRIMARY KEY,
  user_id UUID REFERENCES profiles(id),
  service_id INTEGER REFERENCES services(id),
  product_id INTEGER REFERENCES products(id),
  booking_date DATE,
  booking_time TIME,
  district TEXT,
  address TEXT,
  status TEXT DEFAULT 'pending',
  total_amount_sar DECIMAL,
  language_preference TEXT DEFAULT 'en', -- User's language at booking time
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE messages (
  id SERIAL PRIMARY KEY,
  booking_id INTEGER REFERENCES bookings(id),
  original_text TEXT,
  original_language TEXT, -- 'ar', 'en'
  translated_text TEXT,   -- Bangla translation
  created_at TIMESTAMP DEFAULT NOW()
);
